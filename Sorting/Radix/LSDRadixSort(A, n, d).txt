procedure LSDRadixSort(A, n, d)
    for k=0,1,...,d-1 do
        B <- Array of 10 empty lists
        for i=0,1,...,n-1 do
            digit <- kth digit of A[i]
            Add A[i] to the end of the list B[digit]
        end for
        S <- Empty list
        for j=0,1,...,9 do
            Append B[j] to the end of S
        end for
        Copy S into A
    end for
end procedure
